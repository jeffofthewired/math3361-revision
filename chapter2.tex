        % Part (a)
        \item \textcolor{question}{
            $(Y_n)_{n\in \mathbb N}$ given by 
            $Y_n = \sum_{i=0}^n X_i$, where $X_i$ are \iid\ random variables.
        }

        Let $\mathbb F$ be the natural filtration of $Y$, so $\mathcal F_t = \sigma (Y_s:s\leq t)$.

        \textbf{Showing that $F_t=\sigma(X_s:s\leq t)$}

        For all $t\in \mathbb N$, the random vector $(X_1,\dots,X_t)$ 
        can be expressed as a function of $(Y_1,\dots,Y_t)$, and vice versa.
        \begin{align*}
            \begin{pmatrix}X_1\\X_2\\\vdots\\X_t\end{pmatrix}
            \mapsto \begin{pmatrix}X_1\\ X_1 + X_2 \\\vdots\\ \sum_{i=1}^n X_i\end{pmatrix}\qquad\text{and}\qquad
            \begin{pmatrix}Y_1\\Y_2\\\vdots\\Y_t\end{pmatrix}
            \mapsto \begin{pmatrix}Y_1\\ Y_2 - Y_1\\\vdots\\ Y_n - Y_{n-1}\end{pmatrix}
        \end{align*}
        Thus the $\sigma$-algebras $\mathcal F_t = \sigma(Y_s:s\leq t)$ and $\sigma(X_s:s\leq t)$ are equivalent.

        \textbf{Showing that $\sum_{i=s+1}^t X_i$ is independent to $F_s$}

        We know that the random vector $(X_1,\dots, X_s)$ is independent to $\sigma(X_{s+1},\dots, X_t)$.\\
        Since $\sum_{i=s+1}^t X_i$ is a function of $(X_1,\dots, X_s)$, it is also independent to $\sigma(X_{s+1},\dots, X_t)$.

        \textbf{Proving Markov Property}

        Let $f:\mathbb{R}\to\mathbb{R}$ be any function. Then the conditional expectation
        \begin{align*}
            \mathbb E\left[f\left(Y_t\right)\middle|\mathcal F_s\right] 
            &= \mathbb E\left[f\left(Y_s + \sum_{i=s+1}^t X_i \right)\middle|\mathcal F_s\right]\\
            &= \mathbb E\left[f\left(y + \sum_{i=s+1}^t X_i \right)\right]\bigg|_{y=Y_s},
        \end{align*}
        which only depends on $\omega$ through $Y_s$, and thus is $\sigma(Y_s)$-measurable.\\
        Therefore $Y$ is $\mathbb F$-Markov.

        % Part (b)
        \item \textcolor{question}{
            $B_t=\frac{1}{2}(W_{t+a}-W_t)$, where $a>0$, and $W$ is 
            a Wiener process with natural filtration $\mathbb F$.
        }

        The process $B_t$ is not adapted to $\mathbb F$ \textcolor{red}{(why?)} and therefore cannot be $\mathbb F$-Markov.

        To salvage the quesiton, let $\mathbb F'$ be the filtration shifted by $a$
        $$\mathcal F'_t = \mathcal F_{t+a}.$$

        For $B_t$ to be $\mathbb F'$-Markov, for all functions $f:\mathbb R\to\mathbb R$,
        and for all $0\leq s\leq t\leq T$, \\the conditional expectation
        $\mathbb E\left[f(B_t)\middle|\mathcal F_s'\right]$
        must be $\sigma(B_s)$-measurable.

        Suppose that $t = s + r$, where $0 < r < a$.
        \begin{align*}
            \mathbb E\left[f\!\left(B_{s+r}\right)\middle| \mathcal F_s'\right]
            &= \mathbb E\left[f\!\left(\frac{1}{2}(W_{s+r+a}-W_{s+r})\right)\middle| \mathcal F_{s+a}\right]\\
            &= \mathbb E\left[f\!\left(\frac{1}{2}(W_{s+a+r}-W_{s+a} + W_{s+a} - W_{s+r})\right)\middle| \mathcal F_{s+a}\right]
        \end{align*}
        Note that the Wiener increment $W_{s+a+r}-W_{s+a}$ is independent of $\mathcal F_{s+a}$,\\
        and the Wiener increment $W_{s+a} - W_{s+r}$ is $\mathcal F_{s+a}$-measurable. Thus
        \begin{align*}
            \mathbb E\left[f\!\left(B_{s+r}\right)\middle| \mathcal F_s'\right]
            &= \mathbb E\left[f\!\left(\frac{1}{2}(W_{s+a+r}-W_{s+a} + x)\right)\right]\bigg|_{x=W_{s+a} - W_{s+r}},
        \end{align*}
        which is some function of the random variable $W_{s+a} - W_{s+r}$.

        Our claim is that $B$ is not $\mathbb F'$-Markov, so we just need to 
        find one example of when the resulting function is not $\sigma(B_s)$-measurable.
        Let $f=x\mapsto 2x$, then
        \begin{align*}
            \mathbb E\left[f\!\left(B_{s+r}\right)\middle| \mathcal F_s'\right]
            &= \mathbb E\left[W_{s+a+r}-W_{s+a} + x\right]\big|_{x=W_{s+a} - W_{s+r}}\\
            &= \mathbb E\left[W_{s+a+r}-W_{s+a}\right] + x\big|_{x=W_{s+a} - W_{s+r}}\\
            &= W_{s+a} - W_{s+r}\\
            &= (W_{s+a} - W_s) - (W_{s+r} - W_s)\\
            &= B_s - (W_{s+r} - W_s).
        \end{align*}
        Since $W_{s+r} - W_s$ is not $B_s$-measurable \textcolor{red}{(why?)}, the conditional expectation 
        on the LHS is also non-measurable,
        and thus $B$ is not $\mathbb F'$-Markov.
